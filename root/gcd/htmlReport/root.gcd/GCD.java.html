package root.gcd;

/**
 * @author Victor Kuliamin
 *
 */
public class GCD
{
    public int gcd(int x, int y)
    {
        if (x == -2147483648 && y != 0) {
            x %= y;
        } else if (y == -2147483648 && x != 0) {
            y %= x;
        }

        int t;

        if(x < 0) x = -x;
        if(y < 0) y = -y;

        while(y != 0)
        {
            if(y > x)
            {
                x = y-x;
                y = y-x;
                x = x+y;
            }

            if(y == 0) return x;

            t = y;
            y = x%y;
            x = t;
        }
        return x;
    }


}
